\begin{equation*} A = \left[\begin{array}{rr} 1 \amp 2 \\ 2 \amp 1 \\ \end{array}\right],\qquad D = \left[\begin{array}{rr} 3 \amp 0 \\ 0 \amp -1 \\ \end{array}\right]\text{.} \end{equation*}
